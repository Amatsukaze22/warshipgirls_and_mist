event = { #每年用伪随机概率判定堕落妈触发事件
	id = wg_origin.5
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		exists = event_target:wg_fallen_empire
		any_country = {
			has_country_flag = origin_wg_fe_child_flag
		}
		end_game_years_passed < 0
		years_passed >= 0
	}

	immediate = {
		every_playable_country = {
			limit = {
				is_ai = no
				has_country_flag = origin_wg_fe_child_flag
			}
			random_list = {
				1 = {
					modifier = {
						add = 2
						check_variable = { which = years_not_trigger_fe_event value >= 8 }
					}
					set_variable = { which = years_not_trigger_fe_event value = 0 }
					country_event = { id = wg_origin.6 random = 359 }
				}
				19 = {
					modifier = {
						add = -2
						check_variable = { which = years_not_trigger_fe_event value >= 1 }
					}
					modifier = {
						add = -4
						check_variable = { which = years_not_trigger_fe_event value >= 2 }
					}
					modifier = {
						add = -5
						check_variable = { which = years_not_trigger_fe_event value >= 4 }
					}
					modifier = {
						add = -7
						check_variable = { which = years_not_trigger_fe_event value >= 6 }
					}
					modifier = {
						factor = 0
						check_variable = { which = years_not_trigger_fe_event value >= 10 }
					}
					change_variable = { which = years_not_trigger_fe_event value = 1 }
				}
			}
		}
	}

}
