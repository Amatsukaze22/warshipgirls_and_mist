create_sh_fe_fleet = {
	if = {
		limit = {
			owner = { has_authority = auth_sh_fe }
		}
		create_fleet = {
			name = "NAME_SH_FE_CAPITAL_FLEET"
			effect = {
                set_owner = event_target:sh_fallen_empire
                while = {
                    count = 6
                    create_ship = {
                        name = random
                        design = "NAME_SH_FE_TITAN"
                        graphical_culture = "fallen_empire_04"
                        upgradable = no
                    }
                }
                while = {
                    count = 40
                    create_ship = {
                        name = random
                        design = "NAME_SH_FE_BATTLE_CRUISE"
                        graphical_culture = "fallen_empire_04"
                        upgradable = no
                    }
                }
                while = {
                    count = 25
                    create_ship = {
                        name = random
                        design = "NAME_SH_FE_ESCORTE"
                        graphical_culture = "fallen_empire_04"
                        upgradable = no
                    }
                }					
                set_location = {
                    target = PREV
                    distance = 5
                    angle = random
                }
                set_fleet_stance = aggressive
                set_fleet_bombardment_stance = Galo
                assign_leader = last_created_leader
                #set_aggro_range_measure_from = self
                #set_aggro_range = 500
            }	
        	settings = { 
            	can_upgrade = no 
            	uses_naval_capacity = yes
            }
		}						
	}					
}
